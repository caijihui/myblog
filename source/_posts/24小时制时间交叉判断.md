## javascript 24小时交叉比较函数

```bash
    function CompareTime(beginTime,endTime,nowTimeStar,nowTimeEnd)
    {

        var strb = beginTime.split (":");
        if (strb.length != 2) {
            return false;
        }

        var stre = endTime.split (":");
        if (stre.length != 2) {
            return false;
        }

        var strns = nowTimeStar.split (":");
        if (strns.length != 2) {
            return false;
        }

        var strne = nowTimeEnd.split (":");
        if (strne.length != 2) {
            return false;
        }

        var b = new Date ();
        var e = new Date ();
        var ns = new Date ();
        var ne = new Date ();

        b.setHours (strb[0]);
        b.setMinutes (strb[1]);
        e.setHours (stre[0]);
        e.setMinutes (stre[1]);
        ns.setHours (strns[0]);
        ns.setMinutes (strns[1]);
        ne.setHours (strne[0]);
        ne.setMinutes (strne[1]);

        //交叉了
        if((ns.getTime () - b.getTime () > 0 && ns.getTime () - e.getTime () < 0 )|| (ne.getTime () - b.getTime () > 0 && ne.getTime () - e.getTime () < 0) || (ns.getTime() <= b.getTime() && ne.getTime() >= e.getTime())){
            return true;
        }

    }
```
